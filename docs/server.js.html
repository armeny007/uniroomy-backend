/**
 * @module server
 *
 * @description
 * Server module exposes API endpoints using Express.js
 *
 */

const fs = require('fs');
const path = require('path');
const https = require('https');
const app = require('express')();
const bodyParser = require('body-parser');
let cors = require('cors');
require('dotenv').config();
const expressGraphQL = require('express-graphql');
const mongoose = require('mongoose');
// const { graphql } = require('graphql');

// needed to use findOneAndUpdate() function
mongoose.set('useFindAndModify', false);

const schema = require('./db/modules/graphql-schema');
// const { initPersonTriggers } = require('./db/modules/person-auth-init');
const authController = require('./controllers/auth-controller');
// const { Person } = require('./db/modules/mongoose-schema');

// Any mongoose schema triggers should be initialized before models are compiled
// initPersonTriggers();
// initModels();
// console.log({ Person });

// let connectionString = 'mongodb://127.0.0.1:27017/uniroomy';
const connectionString = process.env.MONGODB_CONNECTION_STRING;

mongoose.Promise = global.Promise;
mongoose.connect(connectionString, { useNewUrlParser: true, useCreateIndex: true })
  .then(async () => {
    console.log('Connected to database');
    // let st = await Student.find().limit(2);
    // let pe = await Person.find().limit(2);
    // console.log({st});
    // console.log({pe});
    // const result = await graphql(schema, '{ roles { _id name } }');
    // console.log({ result });
  })
  .catch(err => console.log(`Error: ${err}`));

app.use(bodyParser.json());
const corsOptions = {
  origin: '*',
  optionsSuccessStatus: 200, // some legacy browsers (IE11, various SmartTVs) choke on 204 
};
app.use(cors(corsOptions));


app.use('/api/graphiql', expressGraphQL({
  graphiql: true,
  schema,
}));

app.use('/api', authController);

app.use('/api', expressGraphQL({
  graphiql: false,
  schema,
}));

app.use((err, req, res, next) => {
  if (err.status === 404) {
    return res.status(400).json({ error: 'Error 404. Path not found' });
  }
  if (err.status === 500) {
    return res.status(500).json({ error: 'Error 500. Internal server error' });
  }

  next();
});

const port = process.env.PORT || 3003;
// app.listen(port, () => console.log(`Server is running on port ${port}`));

const options = {
  key: fs.readFileSync(path.join(process.env.SSL_PATH, 'privkey.pem')),
  cert: fs.readFileSync(path.join(process.env.SSL_PATH, 'fullchain.pem')),
};

const server = https.createServer(options, app);
server.listen(port, () => {
  console.log(`UniRoomy backend server listening on port ${port}`);
});
