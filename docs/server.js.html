server.js

/**
 * @module server
 *
 * @description
 * Server module exposes API endpoints using Express.js
 *
 */

const fs = require('fs');
const path = require('path');
const https = require('https');
const { app } = require('./app');

const port = process.env.PORT || 3003;
// app.listen(port, () => console.log(`Server is running on port ${port}`));

const options = {
  key: fs.readFileSync(path.join(process.env.SSL_PATH, 'privkey.pem')),
  cert: fs.readFileSync(path.join(process.env.SSL_PATH, 'fullchain.pem')),
};
const server = https.createServer(options, app);
server.listen(port, () => {
  console.log(`UniRoomy backend server listening on port ${port}`);
});
